// Function to parse tasks from the workstream plan
function parseWorkstreamPlan(content: string): string[] {
    const tasks: string[] = [];
    const taskRegex = /-\s\[\s\]\s(.*?)\n\s+`added_by:/g;
    let match;
    while ((match = taskRegex.exec(content)) !== null) {
        tasks.push(match[1].trim());
    }
    return tasks;
}
